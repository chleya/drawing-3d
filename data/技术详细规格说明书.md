# NeuralSite智网六维时空AI施工管理系统

## 技术详细规格说明书 v1.0

**可直接交给程序员编程的详细技术规格**

---

## 第一部分：技术选型

### 1.1 编程语言与框架

| 层次 | 技术选型 | 理由 |
|------|----------|------|
| **移动端（现场使用）** | Flutter 3.x | 跨平台（iOS/Android）、高性能、本地数据库支持 |
| **Web管理端** | React 18 + TypeScript | 生态丰富、Cesium/Three.js集成好 |
| **后端API** | Go (Gin框架) | 高并发、性能好，适合工程现场高并发场景 |
| **AI服务** | Python 3.10+ (FastAPI) | PyTorch/LangChain生态最强 |
| **3D/GIS引擎** | CesiumJS + Three.js | WebGIS最佳方案 |

### 1.2 数据库选型

| 用途 | 技术选型 | 理由 |
|------|----------|------|
| **核心业务数据** | PostgreSQL 15 + PostGIS | 空间数据支持最强 |
| **图纸知识图谱** | Neo4j | 图查询效率最高 |
| **非结构化数据** | MongoDB | 文档存储灵活 |
| **缓存/排行榜** | Redis | Sorted Set适合排行榜 |
| **向量检索** | pgvector | 智能问答向量匹配 |

### 1.3 基础设施

- **容器化**: Docker + Kubernetes
- **消息队列**: RabbitMQ（任务队列）
- **对象存储**: MinIO（私有化部署）
- **部署环境**: 支持私有化/云端混合部署

## 第二部分：数据库设计

### 2.1 核心表结构（PostgreSQL）

```sql
-- =============================================
-- 项目管理模块
-- =============================================

-- 项目表
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(200) NOT NULL,
    code VARCHAR(50) UNIQUE NOT NULL,  -- 项目编号
    region VARCHAR(100),                -- 区域
    start_date DATE,                    -- 开工日期
    end_date DATE,                      -- 计划完工日期
    total_budget DECIMAL(18,2),        -- 总预算
    status VARCHAR(20) DEFAULT 'planning',  -- planning/ongoing/completed
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 标段表（道路工程特有）
CREATE TABLE sections (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),
    name VARCHAR(100),                  -- 标段名称
    start_stake VARCHAR(50),           -- 起点桩号，如K0+000
    end_stake VARCHAR(50),              -- 终点桩号，如K5+000
    length DECIMAL(12,2),              -- 长度（米）
    contractor VARCHAR(200),            -- 施工单位
    status VARCHAR(20) DEFAULT 'pending'
);

-- =============================================
-- 空间点位模块（核心：图纸→空间映射）
-- =============================================

-- 空间点位表（每个需要管理的施工位置）
CREATE TABLE spatial_points (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),
    section_id UUID REFERENCES sections(id),

    -- 空间坐标（核心字段）
    stake VARCHAR(50),                 -- 里程桩号，如K5+800
    offset DECIMAL(10,2),              -- 距中桩偏移（米）
    elevation DECIMAL(10,3),          -- 设计高程（米）

    -- 坐标系统一（用于GIS显示）
    longitude DECIMAL(12,8),          -- 经度
    latitude DECIMAL(12,8),           -- 纬度
    height DECIMAL(10,2),             -- 高程

    -- 语义信息
    point_type VARCHAR(50),            -- point_type: road_base/pavement/bridge_pier/culvert等
    name VARCHAR(200),                 -- 点位名称
    description TEXT,
    design_data JSONB,                -- 设计参数JSON

    created_at TIMESTAMP DEFAULT NOW()
);

-- 空间索引（加速空间查询）
CREATE INDEX idx_spatial_points_location
    ON spatial_points USING GIST (ST_MakePoint(longitude, latitude));

-- =============================================
-- 任务与进度模块（变量驱动核心）
-- =============================================

-- 施工任务表
CREATE TABLE tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),
    spatial_point_id UUID REFERENCES spatial_points(id),

    title VARCHAR(200) NOT NULL,
    description TEXT,
    task_type VARCHAR(50),            -- task_type: earthwork/pavement/bridge/culvert

    -- 计划时间
    planned_start DATE,
    planned_end DATE,

    -- 实际时间
    actual_start DATE,
    actual_end DATE,

    -- 状态与进度
    status VARCHAR(20) DEFAULT 'pending',  -- pending/in_progress/completed/delayed
    progress INTEGER DEFAULT 0,         -- 0-100百分比

    -- 资源需求（变量驱动）
    labor_requirement JSONB,          -- {"普通工":10, "技术工":5}
    material_requirement JSONB,       -- {"碎石":500吨, "水泥":100吨}
    equipment_requirement JSONB,      -- {"挖掘机":2, "压路机":1}

    -- 约束条件
    weather_constraint JSONB,          -- {"禁止雨天":true, "温度范围":[10,35]}

    created_by UUID,
    assigned_to UUID[],
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 任务执行记录（用于进度追踪）
CREATE TABLE task_executions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id UUID REFERENCES tasks(id),

    execution_date DATE NOT NULL,
    progress_change INTEGER,           -- 当日进度变化
    work_done TEXT,                   -- 完成工作内容
    problems TEXT,                    -- 存在问题
    photos TEXT[],                    -- 现场照片URL数组

    recorded_by UUID,
    recorded_at TIMESTAMP DEFAULT NOW()
);

-- =============================================
-- 资源变量模块（变量驱动核心）
-- =============================================

-- 人员表
CREATE TABLE workers (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),

    name VARCHAR(50) NOT NULL,
    id_card VARCHAR(18),              -- 身份证号
    phone VARCHAR(11),

    -- 岗位信息
    role VARCHAR(50),                 -- role: manager/engineer/worker
    skill_level INTEGER DEFAULT 1,    -- 技能等级1-5
    specialty VARCHAR(50),           -- 工种：钢筋工/模板工/混凝土工等
    team_id UUID,                     -- 所属班组

    -- 定位（用于现场追踪）
    last_location GEOMETRY(POINT),
    last_location_time TIMESTAMP,

    -- 游戏化积分
    xp INTEGER DEFAULT 0,            -- 经验值
    gold INTEGER DEFAULT 0,          -- 金币
    honor INTEGER DEFAULT 0,         -- 荣誉值
    level INTEGER DEFAULT 1,         -- 等级

    status VARCHAR(20) DEFAULT 'active',  -- active/inactive
    created_at TIMESTAMP DEFAULT NOW()
);

-- 设备表
CREATE TABLE equipment (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),

    name VARCHAR(100) NOT NULL,
    equipment_type VARCHAR(50),      -- equipment_type: excavator/bulldozer/paver等
    model VARCHAR(50),               -- 型号
    license_plate VARCHAR(20),       -- 车牌号

    -- 状态
    status VARCHAR(20) DEFAULT 'available',  -- available/in_use/maintenance
    current_location GEOMETRY(POINT),

    -- 运行数据
    hourly_fuel DECIMAL(6,2),        -- 小时油耗
    total_hours DECIMAL(10,2),       -- 总运行时长

    created_at TIMESTAMP DEFAULT NOW()
);

-- 材料库存表
CREATE TABLE materials (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),

    name VARCHAR(100) NOT NULL,
    material_type VARCHAR(50),        -- material_type: stone/cement/asphalt等
    unit VARCHAR(20),                -- 单位：吨/立方米/块

    -- 库存
    quantity DECIMAL(12,2),          -- 当前库存
    reserved_quantity DECIMAL(12,2), -- 预留数量
    available_quantity DECIMAL(12,2), -- 可用数量

    -- 位置
    storage_location VARCHAR(200),    -- 存放地点

    updated_at TIMESTAMP DEFAULT NOW()
);

-- =============================================
-- 质量管理模块
-- =============================================

-- 质量检查记录
CREATE TABLE quality_inspections (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),
    spatial_point_id UUID REFERENCES spatial_points(id),
    task_id UUID REFERENCES tasks(id),

    inspection_type VARCHAR(50),     -- inspection_type: routine/acceptance/special
    result VARCHAR(20),               -- result: qualified/unqualified

    -- 检查内容
    check_items JSONB,                -- [{"item":"压实度","standard":96,"actual":95.5,"result":"qualified"}]
    photos TEXT[],
    notes TEXT,

    inspector_id UUID,
    inspected_at TIMESTAMP DEFAULT NOW()
);

-- 质量问题与整改
CREATE TABLE quality_issues (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),
    spatial_point_id UUID REFERENCES spatial_points(id),

    description TEXT NOT NULL,
    severity VARCHAR(20),             -- severity: minor/major/critical
    status VARCHAR(20) DEFAULT 'open',  -- open/in_progress/closed

    -- 整改信息
    required_action TEXT,            -- 要求整改内容
    assigned_to UUID,                -- 整改责任人
    due_date DATE,                   -- 整改期限

    -- 整改结果
    rectification_photos TEXT[],
    rectification_note TEXT,
    closed_by UUID,
    closed_at TIMESTAMP,

    created_at TIMESTAMP DEFAULT NOW()
);

-- =============================================
-- 环境变量模块
-- =============================================

-- 天气记录
CREATE TABLE weather_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id),

    record_date DATE NOT NULL,

    -- 天气数据
    weather_type VARCHAR(20),         -- sunny/cloudy/rainy/snowy
    temperature_high DECIMAL(5,1),   -- 最高温度
    temperature_low DECIMAL(5,1),    -- 最低温度
    wind_speed DECIMAL(5,1),         -- 风速（米/秒）
    precipitation DECIMAL(6,2),      -- 降水量（毫米）

    -- 预报数据
    forecast JSONB,                  -- 未来7天预报

    created_at TIMESTAMP DEFAULT NOW()
);

-- =============================================
-- 游戏化模块
-- =============================================

-- 用户成就
CREATE TABLE achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    code VARCHAR(50) UNIQUE NOT NULL,  -- 成就代码，如"safety_eye"
    description TEXT,
    icon VARCHAR(200),                -- 图标URL
    xp_reward INTEGER DEFAULT 0,      -- 奖励经验值
    gold_reward INTEGER DEFAULT 0,    -- 奖励金币

    condition JSONB NOT NULL         -- 触发条件，如{"type":"issue_found","count":10}

-- 用户成就记录
CREATE TABLE user_achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES workers(id),
    achievement_id UUID REFERENCES achievements(id),
    earned_at TIMESTAMP DEFAULT NOW()

-- 排行榜（使用Redis实现，此处仅作记录）
-- 周排行榜: leaderboard:weekly:{project_id}
-- 月排行榜: leaderboard:monthly:{project_id}
```

### 2.2 图数据库模型（Neo4j）

```cypher
// =============================================
// 图纸知识图谱 Schema
// =============================================

// 节点类型
// (:DesignElement) - 设计元素，如结构层、构件
// (:Material) - 材料
// (:Standard) - 规范条款
// (:Location) - 位置（桩号）
// (:Process) - 施工工艺

// 创建示例图谱

// 1. 道路结构层关系
CREATE (p:DesignElement {name: "上面层", type: "pavement_layer", thickness: 40, unit: "mm"})
CREATE (m:Material {name: "AC-13C沥青混凝土", type: "asphalt_concrete"})
CREATE (loc:Location {stake: "K5+800", range: "K5+000-K6+000"})
CREATE (proc:Process {name: "沥青面层摊铺", temperature: "160-180", speed: "2-3m/min"})

CREATE (p)-[:MADE_OF]->(m)
CREATE (p)-[:LOCATED_AT]->(loc)
CREATE (p)-[:PROCESSED_BY]->(proc)

// 2. 材料规格关系
CREATE (m)-[:MUST_FOLLOW]->(s:Standard {code: "JTG F40-2004", name: "公路沥青路面施工技术规范"})

// 3. 结构层上下层关系
CREATE (p2:DesignElement {name: "下面层", type: "pavement_layer", thickness: 60})
CREATE (p)-[:BELOW]->(p2)
CREATE (p2)-[:ABOVE]->(p)

// 查询示例：查询K5+800处路面结构
MATCH (loc:Location {stake: "K5+800"})<-[:LOCATED_AT]-(layer:DesignElement)
<-[:BELOW*0..]-(allLayers)
RETURN layer, allLayers
ORDER BY layer.thickness
```

```dart
// Flutter端同步服务伪代码
class SyncService {
    final LocalDatabase _localDb;
    final RemoteApi _remoteApi;
    final StreamController<SyncStatus> _statusController;

    // 待同步队列
    final List<ChangeRecord> _pendingChanges = [];

    // 初始化
    Future<void> init() async {
        // 1. 加载本地待同步数据
        _pendingChanges = await _localDb.getUnsyncedChanges();

        // 2. 监听网络状态
        connectivity.onConnectivityChanged.listen((status) {
            if (status == ConnectivityResult.wifi ||
                status == ConnectivityResult.mobile) {
                _triggerSync();
            }
        });
    }

    // 记录变更
    Future<void> recordChange(Table table, String operation,
        Map<String, dynamic> data) async {
        // 1. 先保存到本地
        await _localDb.applyChange(table, operation, data);

        // 2. 加入待同步队列
        _pendingChanges.add(ChangeRecord(
            table: table,
            operation: operation,
            data: data,
            localTimestamp: DateTime.now()
        ));

        // 3. 尝试立即同步（如果在线）
        if (await connectivity.isConnected()) {
            _triggerSync();
        }
    }

    // 执行同步
    Future<void> _triggerSync() async {
        if (_pendingChanges.isEmpty) return;

        _statusController.add(SyncStatus.syncing);

        try {
            // 1. 获取服务端自上次同步以来的变更
            final pullChanges = await _remoteApi.pullChanges(
                lastSync: _lastSyncTime
            );

            // 2. 应用服务端变更到本地（解决冲突）
            await _applyPullChanges(pullChanges);

            // 3. 上传本地变更
            final pushResult = await _remoteApi.pushChanges(
                changes: _pendingChanges
            );

            // 4. 处理冲突
            if (pushResult.conflicts.isNotEmpty) {
                await _resolveConflicts(pushResult.conflicts);
            }

            // 5. 标记已同步
            await _localDb.markAsSynced(_pendingChanges);
            _pendingChanges.clear();

            _lastSyncTime = DateTime.now();
            _statusController.add(SyncStatus.synced);

        } catch (e) {
            _statusController.add(SyncStatus.error);
            // 等待后重试
            await Future.delayed(Duration(seconds: 30));
            _triggerSync();
        }
    }

    // 冲突解决策略
    Future<void> _resolveConflicts(List<Conflict> conflicts) async {
        for (final conflict in conflicts) {
            // 策略：服务端时间戳优先（最后写入胜出）
            // 对于重要数据，可提示用户确认
            if (conflict.table == 'tasks' &&
                conflict.criticalField == 'status') {
                // 任务状态冲突：需要用户确认
                _showConflictDialog(conflict);
            } else {
                // 其他字段：自动使用服务端数据
                await _localDb.resolveWithServerData(conflict);
            }
        }
    }
}
```

```python
from fastapi import FastAPI, UploadFile, File
from pydantic import BaseModel
import ezdxf
import re
from neo4j import GraphDatabase

app = FastAPI()

class BlueprintParser:
    def __init__(self, neo4j_driver):
        self.driver = neo4j_driver

    async def parse_dwg(self, file_path: str, project_id: str):
        """解析DWG图纸"""

        # 1. 读取DXF文件
        doc = ezdxf.readfile(file_path)

        # 2. 提取图元
        entities = []
        for entity in doc.modelspace():
            entities.append({
                'type': entity.dxftype(),
                'layer': entity.dxf.layer,
                'color': entity.dxf.color if hasattr(entity.dxf, 'color') else None,
                'coordinates': self._extract_coordinates(entity)
            })

        # 3. 识别实体（使用规则+ML）
        design_elements = []
        for entity in entities:
            element_type = self._classify_entity(entity)
            if element_type:
                design_elements.append({
                    'type': element_type,
                    'layer': entity['layer'],
                    'coordinates': entity['coordinates'],
                    'attributes': self._extract_attributes(entity)
                })

        # 4. 提取里程桩号
        stakes = self._extract_stakes(doc)

        # 5. 构建知识图谱
        await self._build_knowledge_graph(project_id, design_elements, stakes)

        return {
            'status': 'success',
            'elements_count': len(design_elements),
            'stakes_count': len(stakes)
        }

    def _classify_entity(self, entity: dict) -> str:
        """实体分类（规则+ML）"""
        layer = entity.get('layer', '').lower()
        coords = entity.get('coordinates', [])

        # 规则匹配
        if 'pavement' in layer or '路面' in layer:
            return 'pavement_layer'
        elif 'pier' in layer or '墩' in layer:
            return 'bridge_pier'
        elif 'culvert' in layer or '涵' in layer:
            return 'culvert'
        elif 'stake' in layer or '桩' in layer:
            return 'stake_marker'

        # TODO: ML模型分类（预留）
        return None

    def _extract_stakes(self, doc) -> list:
        """提取里程桩号"""
        stakes = []
        text_entities = [e for e in doc.modelspace()
                       if e.dxftype() == 'TEXT']

        pattern = r'[KZ](\d+)\+(\d+)'
        for entity in text_entities:
            text = entity.dxf.text
            matches = re.findall(pattern, text)
            for match in matches:
                stake = f"K{match[0]}+{int(match[1]):03d}"
                stakes.append({
                    'stake': stake,
                    'x': entity.dxf.insert[0] if hasattr(entity.dxf, 'insert') else None,
                    'y': entity.dxf.insert[1] if hasattr(entity.dxf, 'insert') else None
                })

        return stakes

    async def _build_knowledge_graph(self, project_id: str,
                                     elements: list, stakes: list):
        """构建知识图谱"""
        with self.driver.session() as session:
            # 创建项目节点
            await session.run(
                "CREATE (p:Project {id: $project_id})",
                project_id=project_id
            )

            # 创建设计元素节点
            for elem in elements:
                await session.run("""
                    MATCH (p:Project {id: $project_id})
                    CREATE (e:DesignElement {
                        type: $type,
                        layer: $layer,
                        coordinates: $coords,
                        attributes: $attrs
                    })
                    CREATE (p)-[:HAS_ELEMENT]->(e)
                """,
                project_id=project_id,
                type=elem['type'],
                layer=elem['layer'],
                coords=str(elem['coordinates']),
                attrs=str(elem['attributes'])
                )

            # 建立桩号与位置的关联
            for stake in stakes:
                await session.run("""
                    MATCH (e:DesignElement {project_id: $project_id})
                    WHERE e.coordinates CONTAINS $approx_coords
                    CREATE (l:Location {stake: $stake})
                    CREATE (e)-[:LOCATED_AT]->(l)
                """,
                project_id=project_id,
                stake=stake['stake'],
                approx_coords=str(stake['x'])[:4]  # 近似匹配
                )

# API端点
```
